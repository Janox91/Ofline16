🔍 Diagnóstico - Control de Vigilancia

Esta página te ayudará a diagnosticar problemas en la aplicación principal.

📋 Instrucciones:

  1. Haz clic en "Ir a la aplicación principal"
  2. Una vez en la aplicación, abre las herramientas de desarrollador (F12)
  3. Ve a la pestaña Console
  4. Ejecuta el comando de diagnóstico que aparece abajo
  5. Comparte los resultados si hay errores

🔧 Código de Diagnóstico:

Copia y pega este código en la consola del navegador (F12 → Console):

// Diagnóstico de elementos del DOM
function checkElement(id, description) {
    const element = document.getElementById(id);
    if (element) {
        console.log(`✅ ${description} (${id}) encontrado`);
        return true;
    } else {
        console.log(`❌ ${description} (${id}) NO encontrado`);
        return false;
    }
}

function checkClass(className, description) {
    const elements = document.querySelectorAll(className);
    if (elements.length > 0) {
        console.log(`✅ ${description} (${className}) encontrado - ${elements.length} elementos`);
        return true;
    } else {
        console.log(`❌ ${description} (${className}) NO encontrado`);
        return false;
    }
}

console.log('🔍 Iniciando diagnóstico de elementos del DOM...');
console.log('='.repeat(50));

const results = [];

// Verificar contenedores principales
results.push(checkClass('.app-container', 'Contenedor principal'));
results.push(checkClass('.app-header', 'Header de la aplicación'));
results.push(checkClass('.app-nav', 'Navegación'));
results.push(checkClass('.app-main', 'Contenido principal'));

// Verificar vistas
results.push(checkElement('dashboard', 'Vista Dashboard'));
results.push(checkElement('attendance', 'Vista Asistencia'));
results.push(checkElement('employees', 'Vista Personal'));
results.push(checkElement('reports', 'Vista Reportes'));
results.push(checkElement('planilla', 'Vista Planilla'));

// Verificar botones principales
results.push(checkElement('addEmployeeBtn', 'Botón Agregar Personal'));
results.push(checkElement('exportBtn', 'Botón Exportar'));
results.push(checkElement('addNewEmployeeBtn', 'Botón Agregar Nuevo Personal'));
results.push(checkElement('generateReportBtn', 'Botón Generar Reporte'));

// Verificar formularios
results.push(checkElement('employeeForm', 'Formulario de Personal'));
results.push(checkElement('attendanceForm', 'Formulario de Asistencia'));

// Verificar inputs
results.push(checkElement('attendanceDate', 'Input Fecha Asistencia'));
results.push(checkElement('reportStartDate', 'Input Fecha Inicio Reporte'));
results.push(checkElement('reportEndDate', 'Input Fecha Fin Reporte'));
results.push(checkElement('planillaMonth', 'Input Mes Planilla'));

// Verificar modales
results.push(checkElement('employeeModal', 'Modal de Personal'));
results.push(checkElement('attendanceModal', 'Modal de Asistencia'));

console.log('='.repeat(50));

// Resumen
const successCount = results.filter(r => r).length;
const totalCount = results.length;
const successRate = Math.round((successCount / totalCount) * 100);

console.log(`📊 Resumen: ${successCount}/${totalCount} elementos encontrados (${successRate}%)`);

if (successRate === 100) {
    console.log('🎉 ¡Todos los elementos están presentes! La aplicación debería funcionar correctamente.');
} else {
    console.log('⚠️ Faltan algunos elementos. Esto podría causar errores en la aplicación.');
}

console.log('🔍 Diagnóstico completado.');
